export class ErrorHandler {
  static handle(error, context = '') {
    console.error(`❌ Ошибка в ${context}:`, error);

    let message = '❌ Неизвестная ошибка';

    if (typeof error === 'string') {
      message = error;
    } else if (error.message) {
      // Supabase ошибки
      if (error.message.includes('relation')) {
        message = '❌ Таблица не найдена в базе';
      } else if (error.message.includes('PGRST')) {
        message = '❌ Ошибка базы данных';
      } else if (error.message.includes('auth')) {
        message = '❌ Ошибка аутентификации';
      } else if (error.message.includes('network')) {
        message = '❌ Нет интернета';
      } else {
        message = `❌ ${error.message}`;
      }
    }

    return {
      message,
      originalError: error,
      context,
      timestamp: new Date().toISOString()
    };
  }

  static getUserMessage(error) {
    const errorObj = this.handle(error);
    return errorObj.message;
  }

  static isNetworkError(error) {
    return error.message?.includes('network') ||
           error.message?.includes('timeout') ||
           error.message?.includes('Failed');
  }

  static isAuthError(error) {
    return error.message?.includes('auth') ||
           error.message?.includes('unauthorized');
  }

  static isValidationError(error) {
    return error.message?.includes('validation') ||
           error.message?.includes('invalid');
  }
}

export class CustomError extends Error {
  constructor(message, code = 'ERROR', details = {}) {
    super(message);
    this.code = code;
    this.details = details;
    this.timestamp = new Date().toISOString();
  }
}

export class NetworkError extends CustomError {
  constructor(message = 'Нет интернета') {
    super(message, 'NETWORK_ERROR');
  }
}

export class AuthError extends CustomError {
  constructor(message = 'Ошибка аутентификации') {
    super(message, 'AUTH_ERROR');
  }
}

export class ValidationError extends CustomError {
  constructor(message = 'Ошибка валидации', details = {}) {
    super(message, 'VALIDATION_ERROR', details);
  }
}

export class NotFoundError extends CustomError {
  constructor(message = 'Не найдено') {
    super(message, 'NOT_FOUND');
  }
}

export default ErrorHandler;
