// ВАЛИДАЦИЯ ДАННЫХ

export class Validation {
  // Проверка номера телефона
  static isValidPhone(phone) {
    const cleaned = phone.replace(/\D/g, '');
    return cleaned.length === 12 && cleaned.startsWith('992');
  }

  // Проверка Email
  static isValidEmail(email) {
    const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return regex.test(email);
  }

  // Проверка имени
  static isValidName(name) {
    return name && name.trim().length >= 2;
  }

  // Проверка цены
  static isValidPrice(price) {
    const num = parseFloat(price);
    return !isNaN(num) && num >= 10;
  }

  // Проверка количества мест
  static isValidSeats(seats) {
    const num = parseInt(seats);
    return !isNaN(num) && num >= 1 && num <= 8;
  }

  // Проверка маршрута
  static isValidRoute(from, to) {
    return from && from.trim().length > 0 && to && to.trim().length > 0;
  }

  // Проверка даты
  static isValidDate(dateString) {
    const date = new Date(dateString);
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    return date >= today;
  }

  // Проверка времени
  static isValidTime(timeString) {
    const regex = /^([0-1]?[0-9]|2[0-3]):[0-5][0-9]$/;
    return regex.test(timeString);
  }

  // Проверка SMS кода
  static isValidSmsCode(code) {
    return /^\d{4}$/.test(code);
  }

  // Проверка координат
  static isValidCoordinates(lat, lng) {
    return (
      !isNaN(lat) &&
      !isNaN(lng) &&
      lat >= -90 &&
      lat <= 90 &&
      lng >= -180 &&
      lng <= 180
    );
  }

  // Проверка марки машины
  static isValidCarModel(model) {
    return model && model.trim().length >= 3;
  }

  // Проверка номерного знака
  static isValidPlateNumber(plate) {
    return plate && plate.trim().length >= 5;
  }

  // Валидация всей формы поездки
  static validateTripForm(formData) {
    const errors = [];

    if (!this.isValidRoute(formData.from, formData.to)) {
      errors.push('❌ Укажите маршрут');
    }

    if (!this.isValidDate(formData.date)) {
      errors.push('❌ Дата не может быть в прошлом');
    }

    if (!this.isValidTime(formData.time)) {
      errors.push('❌ Укажите время');
    }

    if (!this.isValidCarModel(formData.carModel)) {
      errors.push('❌ Укажите марку машины');
    }

    if (!this.isValidPrice(formData.price)) {
      errors.push('❌ Минимальная цена 10 сом');
    }

    if (!this.isValidSeats(formData.seats)) {
      errors.push('❌ Укажите количество мест');
    }

    return {
      isValid: errors.length === 0,
      errors
    };
  }

  // Валидация формы брони
  static validateBookingForm(formData) {
    const errors = [];

    if (!formData.tripId) {
      errors.push('❌ Поездка не выбрана');
    }

    if (!formData.passengerName) {
      errors.push('❌ Укажите имя');
    }

    if (formData.payMethod === 'card') {
      if (!this.isValidCardNumber(formData.cardNumber)) {
        errors.push('❌ Неверный номер карты');
      }
      if (!this.isValidCardExpiry(formData.cardExpiry)) {
        errors.push('❌ Неверный срок карты');
      }
      if (!this.isValidCardCVV(formData.cardCVV)) {
        errors.push('❌ Неверный CVV');
      }
    }

    return {
      isValid: errors.length === 0,
      errors
    };
  }

  // Проверка номера карты
  static isValidCardNumber(cardNumber) {
    const cleaned = cardNumber.replace(/\s/g, '');
    return cleaned.length === 16 && /^\d+$/.test(cleaned);
  }

  // Проверка срока карты
  static isValidCardExpiry(expiry) {
    const regex = /^(0[1-9]|1[0-2])\/\d{2}$/;
    return regex.test(expiry);
  }

  // Проверка CVV
  static isValidCardCVV(cvv) {
    return /^\d{3}$/.test(cvv);
  }
}

export default Validation;
